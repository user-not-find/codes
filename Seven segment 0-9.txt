//Seven segment 0-9
#include <reg51.h>

// Function declarations
void display(unsigned char byte);
void delay(void);

// Main function
void main(void)
{
    P0 = 0x00;
    P1 = 0x04;

    while(1)
    {
        display(0xC0);   // 0
        display(0xF9);   // 1
        display(0xA4);   // 2
        display(0xB0);   // 3
        display(0x99);   // 4
        display(0x92);   // 5
        display(0x82);   // 6
        display(0xF8);   // 7
        display(0x80);   // 8
        display(0x90);   // 9
    }
}

// Display function
void display(unsigned char byte)
{
    P0 = byte;   // Send data to Port 0
    delay();
}

// Delay function
void delay(void)
{
    unsigned int i, j;

    for(i = 0; i < 100; i++)
    {
        for(j = 0; j < 100; j++);
    }
}
