//LED blinking/flashing code

#include<reg51.h> //Header file for 8051 microcontroller

void delay(void);

//Main function
void main(void)

{
 while(1)  //infinite loop to keep LEDs blinking continuously
 {
  P1 = 0xFF; // Turn ON all LEDs connected to Port 1
  delay();

  P1 = 0x00; // Turn OFF all LEDs
  delay();
 }
}

//Function to generate time delay
void delay(void)
{
    int i,j; //loop variables
    for(i=0;i<1000;i++) //outer loop
    {
        for (j=0;j<100;j++) //inner loop
        {
            //empty loop to create delay
        }
    }
}
