//Triangular wave
#include <reg51.h>

void main(void)
{
    unsigned int x;

    while(1)
    {
        // Rising part of triangular wave
        for(x = 0; x < 255; x++)
        {
            P1 = x;   // Ramp up
        }

        // Falling part of triangular wave
        for(x = 255; x > 0; x--)
        {
            P1 = x;   // Ramp down
        }
    }
}
